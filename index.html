// ============================================ // APPS SCRIPT - Amaliyah Kelompok v2 // ============================================ // Sheet "Config" = JSON config di cell A1 // Sheet "Data" = rekap amalan harian (per-cell update) // ============================================ // CARA DEPLOY: // 1. Buka Google Sheets → Extensions → Apps Script // 2. Hapus semua kode lama, paste kode ini // 3. Deploy → New Deployment → Web App // - Execute as: Me // - Who has access: Anyone // 4. Copy URL deployment baru, ganti di index.html // // SETUP AUTO RESET (opsional, bersihkan data >7 hari): // 5. Di Apps Script, klik ikon jam ⏰ (Triggers) di sidebar kiri // 6. + Add Trigger: // - Function: dailyReset // - Event source: Time-driven // - Type: Day timer // - Time of day: Midnight to 1am // 7. Save → Authorize // ============================================ function doGet(e) { var action = (e && e.parameter) ? e.parameter.action : ""; // === NEW: getAll = config + amalan data for today === if (action === "getAll") { var date = e.parameter.date || ""; return getAllData(date); } // === EXISTING: getConfig === if (action === "getConfig") { return getConfig(); } // === EXISTING: saveConfig via GET === if (action === "saveConfig") { var payload = e.parameter.payload; var data = JSON.parse(payload); return saveConfigData(data); } // === EXISTING: saveData via GET (bulk) === if (action === "saveData") { var payload = e.parameter.payload; var data = JSON.parse(payload); return saveAmalanData(data); } return ContentService.createTextOutput("Amaliyah API v2 Ready"); } function doPost(e) { try { var raw = ""; if (e.parameter && e.parameter.payload) { raw = e.parameter.payload; } else if (e.postData && e.postData.contents) { raw = e.postData.contents; } if (!raw) return ContentService.createTextOutput("No data"); var data = JSON.parse(raw); // === NEW: saveAmalan (per-cell real-time save) === if (data.action === "saveAmalan") { return saveAmalanCell(data); } // === EXISTING: saveConfig via POST === if (data.action === "saveConfig") { return saveConfigData(data); } // === EXISTING: bulk save === return saveAmalanData(data); } catch(err) { return ContentService.createTextOutput("Error: " + err.message); } } // ============================================ // NEW: GET ALL DATA (config + amalan for date) // Called on every app open so all phones see same data // ============================================ function getAllData(date) { var ss = SpreadsheetApp.getActiveSpreadsheet(); var result = { config: { members: [], amalan: [] }, data: [] }; // Load config var configSheet = ss.getSheetByName("Config"); if (configSheet) { var raw = configSheet.getRange("A1").getValue(); if (raw) { try { result.config = JSON.parse(raw); } catch(e) {} } } // Load amalan data for specific date var dataSheet = ss.getSheetByName("Data"); if (dataSheet && dataSheet.getLastRow() > 1) { var all = dataSheet.getDataRange().getValues(); // Headers: tanggal | nama | amalanId | amalanName | value | type for (var i = 1; i < all.length; i++) { var rowDate = formatDate(all[i][0]); if (rowDate === date) { result.data.push({ tanggal: rowDate, nama: str(all[i][1]), amalanId: str(all[i][2]), amalanName: str(all[i][3]), value: str(all[i][4]), type: str(all[i][5]) }); } } } return ContentService .createTextOutput(JSON.stringify(result)) .setMimeType(ContentService.MimeType.JSON); } // ============================================ // NEW: SAVE AMALAN CELL (real-time per-click) // Upsert: update existing row or insert new // ============================================ function saveAmalanCell(data) { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheetByName("Data"); // Create sheet + headers if needed if (!sheet) { sheet = ss.insertSheet("Data"); sheet.appendRow(["tanggal", "nama", "amalanId", "amalanName", "value", "type"]); } if (sheet.getLastRow() === 0) { sheet.appendRow(["tanggal", "nama", "amalanId", "amalanName", "value", "type"]); } var tanggal = data.tanggal; var nama = data.nama; var amalanId = data.amalanId; // Search for existing row var found = false; var lastRow = sheet.getLastRow(); if (lastRow > 1) { var rows = sheet.getRange(2, 1, lastRow - 1, 6).getValues(); for (var i = 0; i < rows.length; i++) { var rd = formatDate(rows[i][0]); if (rd === tanggal && rows[i][1] == nama && rows[i][2] == amalanId) { // Update value + type sheet.getRange(i + 2, 5).setValue(data.value); sheet.getRange(i + 2, 6).setValue(data.type); found = true; break; } } } // Insert new row if not found if (!found) { sheet.appendRow([tanggal, nama, amalanId, data.amalanName || "", data.value, data.type || "check"]); } return ContentService.createTextOutput("OK"); } // ============================================ // EXISTING: GET CONFIG // ============================================ function getConfig() { var ss = SpreadsheetApp.getActiveSpreadsheet(); var configSheet = ss.getSheetByName("Config"); if (!configSheet) { return ContentService .createTextOutput(JSON.stringify({ members: [], amalan: [] })) .setMimeType(ContentService.MimeType.JSON); } var data = configSheet.getRange("A1").getValue(); if (!data) { return ContentService .createTextOutput(JSON.stringify({ members: [], amalan: [] })) .setMimeType(ContentService.MimeType.JSON); } try { var config = JSON.parse(data); return ContentService .createTextOutput(JSON.stringify(config)) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService .createTextOutput(JSON.stringify({ members: [], amalan: [] })) .setMimeType(ContentService.MimeType.JSON); } } // ============================================ // EXISTING: SAVE CONFIG // ============================================ function saveConfigData(data) { var ss = SpreadsheetApp.getActiveSpreadsheet(); var configSheet = ss.getSheetByName("Config"); if (!configSheet) { configSheet = ss.insertSheet("Config"); } var config = { members: data.members, amalan: data.amalan }; configSheet.getRange("A1").setValue(JSON.stringify(config)); return ContentService.createTextOutput("Config saved OK"); } // ============================================ // EXISTING: SAVE AMALAN DATA (bulk - from Save button) // ============================================ function saveAmalanData(data) { var ss = SpreadsheetApp.getActiveSpreadsheet(); var dataSheet = ss.getSheetByName("Data"); if (!dataSheet) { dataSheet = ss.insertSheet("Data"); } if (dataSheet.getLastRow() === 0) { dataSheet.appendRow(data.headers); } data.data.forEach(function(row) { var r = data.headers.map(function(h) { return row[h] !== undefined ? row[h] : ""; }); dataSheet.appendRow(r); }); return ContentService.createTextOutput("OK - " + data.data.length + " rows saved"); } // ============================================ // NEW: DAILY RESET // Deletes data older than 7 days to keep sheet clean. // Today's data is automatically "fresh" because the app // only loads rows matching today's date. // // Setup trigger: Triggers → + Add Trigger → // Function: dailyReset // Time-driven → Day timer → Midnight to 1am // ============================================ function dailyReset() { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheetByName("Data"); if (!sheet || sheet.getLastRow() <= 1) return; var cutoff = new Date(); cutoff.setDate(cutoff.getDate() - 7); var rows = sheet.getRange(2, 1, sheet.getLastRow() - 1, 1).getValues(); var toDelete = []; for (var i = 0; i < rows.length; i++) { var d = rows[i][0] instanceof Date ? rows[i][0] : new Date(rows[i][0]); if (d < cutoff) toDelete.push(i + 2); } // Delete from bottom up toDelete.reverse().forEach(function(r) { sheet.deleteRow(r); }); Logger.log("dailyReset: removed " + toDelete.length + " old rows"); } // ============================================ // HELPERS // ============================================ function formatDate(val) { if (!val) return ""; if (val instanceof Date) { var dd = ("0" + val.getDate()).slice(-2); var mm = ("0" + (val.getMonth() + 1)).slice(-2); return val.getFullYear() + "-" + mm + "-" + dd; } return val.toString(); } function str(val) { return val !== undefined && val !== null ? val.toString() : ""; } // ============================================ // TEST FUNCTIONS (run manually) // ============================================ function testGetAll() { var today = Utilities.formatDate(new Date(), "Asia/Jakarta", "yyyy-MM-dd");
